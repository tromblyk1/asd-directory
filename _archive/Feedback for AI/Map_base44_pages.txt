import React, { useState, useMemo, useEffect } from "react";
import { base44 } from "@/api/base44Client";
import { useQuery } from "@tanstack/react-query";
import { Card, CardContent } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { Input } from "@/components/ui/input";
import { 
  MapPin, Phone, Navigation, Search, X 
} from "lucide-react";
import { Link } from "react-router-dom";
import { createPageUrl } from "@/utils";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";

export default function MapPage() {
  const [selectedCategory, setSelectedCategory] = useState("all");
  const [searchTerm, setSearchTerm] = useState("");
  const [selectedResource, setSelectedResource] = useState(null);
  const [MapComponent, setMapComponent] = useState(null);

  const { data: resources = [], isLoading } = useQuery({
    queryKey: ['resources-map'],
    queryFn: () => base44.entities.Resource.list('-created_date', 500),
    initialData: [],
  });

  // Load map dynamically and inject CSS
  useEffect(() => {
    if (typeof window !== 'undefined') {
      // Inject Leaflet CSS
      const link = document.createElement('link');
      link.rel = 'stylesheet';
      link.href = 'https://unpkg.com/leaflet@1.9.4/dist/leaflet.css';
      link.integrity = 'sha256-p4NxAoJBhIIN+hmNHrzRCf9tD/miZyoHS5obTRR9BMY=';
      link.crossOrigin = '';
      document.head.appendChild(link);

      // Load Leaflet React components
      import('react-leaflet').then((module) => {
        setMapComponent(() => module);
      });

      return () => {
        document.head.removeChild(link);
      };
    }
  }, []);

  const filteredResources = useMemo(() => {
    return resources.filter(resource => {
      const hasCoordinates = resource.latitude && resource.longitude;
      const matchesCategory = selectedCategory === "all" || resource.category === selectedCategory;
      const matchesSearch = !searchTerm || 
        resource.name?.toLowerCase().includes(searchTerm.toLowerCase()) ||
        resource.city?.toLowerCase().includes(searchTerm.toLowerCase());
      
      return hasCoordinates && matchesCategory && matchesSearch;
    });
  }, [resources, selectedCategory, searchTerm]);

  const center = useMemo(() => {
    if (filteredResources.length > 0 && filteredResources[0].latitude) {
      return [filteredResources[0].latitude, filteredResources[0].longitude];
    }
    return [27.9944024, -81.7602544]; // Center of Florida
  }, [filteredResources]);

  const clearFilters = () => {
    setSelectedCategory("all");
    setSearchTerm("");
  };

  return (
    <div className="h-screen flex flex-col">
      {/* Header */}
      <div className="bg-gradient-to-r from-blue-600 to-green-600 text-white py-6 px-6">
        <div className="max-w-7xl mx-auto">
          <h1 className="text-3xl font-bold mb-2">Interactive Resource Map</h1>
          <p className="text-blue-50">
            Explore neurodivergent-friendly resources across Florida
          </p>
        </div>
      </div>

      {/* Filters */}
      <div className="bg-white border-b border-gray-200 px-6 py-4">
        <div className="max-w-7xl mx-auto flex flex-col lg:flex-row gap-4">
          <div className="flex-1">
            <div className="relative">
              <Search className="absolute left-3 top-1/2 transform -translate-y-1/2 w-4 h-4 text-gray-400" />
              <Input
                type="text"
                placeholder="Search resources..."
                value={searchTerm}
                onChange={(e) => setSearchTerm(e.target.value)}
                className="pl-10"
              />
            </div>
          </div>
          
          <div className="w-full lg:w-64">
            <Select value={selectedCategory} onValueChange={setSelectedCategory}>
              <SelectTrigger>
                <SelectValue />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">All Categories</SelectItem>
                <SelectItem value="faith_based">Faith Communities</SelectItem>
                <SelectItem value="healthcare">Healthcare</SelectItem>
                <SelectItem value="education">Education</SelectItem>
                <SelectItem value="recreation">Recreation</SelectItem>
                <SelectItem value="support_services">Support Services</SelectItem>
                <SelectItem value="employment">Employment</SelectItem>
                <SelectItem value="daily_living">Daily Living</SelectItem>
              </SelectContent>
            </Select>
          </div>

          {(selectedCategory !== "all" || searchTerm) && (
            <Button variant="outline" onClick={clearFilters}>
              <X className="w-4 h-4 mr-2" />
              Clear
            </Button>
          )}
        </div>
      </div>

      {/* Map and Results */}
      <div className="flex-1 flex flex-col lg:flex-row overflow-hidden">
        {/* Map Container */}
        <div className="flex-1 relative bg-gray-100">
          {isLoading ? (
            <div className="h-full flex items-center justify-center">
              <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-blue-600" />
            </div>
          ) : !MapComponent ? (
            <div className="h-full flex items-center justify-center">
              <p className="text-gray-500">Loading map...</p>
            </div>
          ) : (
            <MapComponent.MapContainer 
              center={center} 
              zoom={7} 
              className="h-full w-full"
              scrollWheelZoom={true}
            >
              <MapComponent.TileLayer
                attribution='&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a>'
                url="https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png"
              />
              {filteredResources.map((resource) => (
                <MapComponent.Marker 
                  key={resource.id}
                  position={[resource.latitude, resource.longitude]}
                  eventHandlers={{
                    click: () => setSelectedResource(resource),
                  }}
                >
                  <MapComponent.Popup>
                    <div className="p-2">
                      <h3 className="font-bold text-sm mb-1">{resource.name}</h3>
                      <p className="text-xs text-gray-600 mb-2">{resource.city}</p>
                      <Link to={createPageUrl("ResourceDetail") + `?id=${resource.id}`}>
                        <Button size="sm" className="text-xs">View Details</Button>
                      </Link>
                    </div>
                  </MapComponent.Popup>
                </MapComponent.Marker>
              ))}
            </MapComponent.MapContainer>
          )}
          
          {/* Results Count Overlay */}
          <div className="absolute top-4 left-4 z-[1000]">
            <Card className="shadow-lg border-none">
              <CardContent className="p-3">
                <p className="text-sm font-medium">
                  <span className="text-blue-600 font-bold">{filteredResources.length}</span> resources
                </p>
              </CardContent>
            </Card>
          </div>
        </div>

        {/* Sidebar - Resource List */}
        <div className="w-full lg:w-96 h-64 lg:h-full overflow-y-auto bg-white border-t lg:border-t-0 lg:border-l border-gray-200">
          <div className="p-4 space-y-3">
            {filteredResources.length === 0 ? (
              <div className="text-center py-12 text-gray-500">
                <MapPin className="w-12 h-12 mx-auto mb-3 text-gray-300" />
                <p>No resources found</p>
              </div>
            ) : (
              filteredResources.map((resource) => (
                <Card 
                  key={resource.id}
                  className={`cursor-pointer hover:shadow-md transition-shadow border ${
                    selectedResource?.id === resource.id ? 'border-blue-500 shadow-md' : ''
                  }`}
                  onClick={() => setSelectedResource(resource)}
                >
                  <CardContent className="p-4">
                    <div className="flex items-start justify-between mb-2">
                      <h3 className="font-semibold text-sm">{resource.name}</h3>
                      <Badge variant="secondary" className="text-xs">
                        {resource.category?.replace(/_/g, ' ')}
                      </Badge>
                    </div>
                    
                    <div className="space-y-1 text-xs text-gray-600">
                      <div className="flex items-center gap-2">
                        <MapPin className="w-3 h-3" />
                        <span>{resource.city}, FL</span>
                      </div>
                      {resource.phone && (
                        <div className="flex items-center gap-2">
                          <Phone className="w-3 h-3" />
                          <span>{resource.phone}</span>
                        </div>
                      )}
                    </div>

                    <div className="flex gap-2 mt-3">
                      <Link 
                        to={createPageUrl("ResourceDetail") + `?id=${resource.id}`}
                        className="flex-1"
                      >
                        <Button size="sm" variant="outline" className="w-full text-xs">
                          View Details
                        </Button>
                      </Link>
                      <a
                        href={`https://www.google.com/maps/dir/?api=1&destination=${resource.latitude},${resource.longitude}`}
                        target="_blank"
                        rel="noopener noreferrer"
                      >
                        <Button size="sm" variant="outline" className="text-xs">
                          <Navigation className="w-3 h-3" />
                        </Button>
                      </a>
                    </div>
                  </CardContent>
                </Card>
              ))
            )}
          </div>
        </div>
      </div>
    </div>
  );
}
